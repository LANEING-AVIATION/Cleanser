操作步骤：
1、远端可访问youtube
2、细节代理配置
3、启动自动化脚本

git config --global http.proxy http://10.238.31.17:7890
git config --global https.proxy https://10.238.31.17:7890

git config --global --unset http.proxy
git config --global --unset https.proxy

mkdir -p ~/.pip
nano ~/.pip/pip.conf

proxy_servers:
  https: http://10.238.31.17:7890
  http: http://10.238.31.17:7890
  

nano ~/.bashrc
export http_proxy="http://10.238.31.17:7890"
export https_proxy="http://10.238.31.17:7890"
source ~/.bashrc


pip install <package_name> --proxy http://10.238.31.17:7890
